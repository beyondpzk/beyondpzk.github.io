# l1与l2正则化的作用

一句话总结: L1 是"剪刀" —— 剪掉不重要的（变 0）; L2 是"弹簧" —— 压缩所有权值（变小但不为 0）

下面是从代码的角度来进行实验分析的, 就假设 `y=ax+b`

```

import matplotlib.pyplot as plt
import numpy as np

np.random.seed(42)

# ============ 1. Generate Simple Data ============
N = 50
x = np.linspace(-3, 3, N)
true_a, true_b = 2.0, -1.0
noise = np.random.randn(N) * 0.5
y = true_a * x + true_b + noise

print(f"Data Generation: True parameters a={true_a}, b={true_b}")
print(f"x shape: {x.shape}, y shape: {y.shape}")

# ============ 2. Define Loss Functions (Fixed Broadcasting) ============

def compute_loss_v3(a, b, x, y):
    """
    Universal version handling both scalar and grid inputs
    Key: Add dimension to (a, b) so broadcasting works correctly
    """
    if np.ndim(a) == 0:
        # Scalar case: direct computation
        y_pred = a * x + b
        return np.mean((y - y_pred) ** 2)
    else:
        # Grid case: add dimension for broadcasting
        # a: (M, K), x: (N,) -> need (M, K, 1) * (N,) = (M, K, N)
        a_exp = a[:, :, np.newaxis]
        b_exp = b[:, :, np.newaxis]
        y_pred = a_exp * x + b_exp  # Broadcasts to (M, K, N)
        # Average over data dimension
        return np.mean((y - y_pred) ** 2, axis=2)

def compute_L1_loss(a, b, x, y, lambda_l1=1.0):
    """L1 regularization loss"""
    return compute_loss_v3(a, b, x, y) + lambda_l1 * (np.abs(a) + np.abs(b))

def compute_L2_loss(a, b, x, y, lambda_l2=1.0):
    """L2 regularization loss"""
    return compute_loss_v3(a, b, x, y) + lambda_l2 * (a**2 + b**2)

# ============ 3. Create Parameter Grid ============
a_range = np.linspace(-2, 6, 200)
b_range = np.linspace(-6, 4, 200)
A, B = np.meshgrid(a_range, b_range)

print(f"A shape: {A.shape}, B shape: {B.shape}")

# Compute losses
Loss_base = compute_loss_v3(A, B, x, y)
Loss_L1 = compute_L1_loss(A, B, x, y, lambda_l1=4.0)
Loss_L2 = compute_L2_loss(A, B, x, y, lambda_l2=4.0)

print(f"Loss shape: {Loss_base.shape}")

# ============ 4. Find Optimal Solutions ============
idx_min = np.argmin(Loss_base)
a_opt_base = a_range[idx_min % 200]
b_opt_base = b_range[idx_min // 200]

idx_min_l1 = np.argmin(Loss_L1)
a_opt_l1 = a_range[idx_min_l1 % 200]
b_opt_l1 = b_range[idx_min_l1 // 200]

idx_min_l2 = np.argmin(Loss_L2)
a_opt_l2 = a_range[idx_min_l2 % 200]
b_opt_l2 = b_range[idx_min_l2 // 200]

print(f"Unregularized optimum: a={a_opt_base:.2f}, b={b_opt_base:.2f}")
print(f"L1 regularized optimum: a={a_opt_l1:.2f}, b={b_opt_l1:.2f}")
print(f"L2 regularized optimum: a={a_opt_l2:.2f}, b={b_opt_l2:.2f}")

# ============ 5. Plot 1: 3D Loss Surface Comparison ============
fig = plt.figure(figsize=(18, 5))

# Unregularized
ax1 = fig.add_subplot(1, 3, 1, projection='3d')
surf1 = ax1.plot_surface(A, B, Loss_base, cmap='viridis',
                         alpha=0.8, edgecolor='none')
ax1.scatter([a_opt_base], [b_opt_base], [Loss_base[idx_min // 200, idx_min % 200]],
            color='red', s=100, label='Optimum')
ax1.set_title('Unregularized Loss', fontweight='bold')
ax1.set_xlabel('Parameter a')
ax1.set_ylabel('Parameter b')
ax1.set_zlabel('Loss')
ax1.legend()

# L1 Regularized
ax2 = fig.add_subplot(1, 3, 2, projection='3d')
surf2 = ax2.plot_surface(A, B, Loss_L1, cmap='viridis',
                         alpha=0.8, edgecolor='none')
ax2.scatter([a_opt_l1], [b_opt_l1], [Loss_L1[idx_min_l1 // 200, idx_min_l1 % 200]],
            color='red', s=100)
ax2.set_title(
    f'L1 Regularization (λ=2)\nOptimum: a={a_opt_l1:.2f}, b={b_opt_l1:.2f}', fontweight='bold')
ax2.set_xlabel('Parameter a')
ax2.set_ylabel('Parameter b')
ax2.set_zlabel('Loss')

# L2 Regularized
ax3 = fig.add_subplot(1, 3, 3, projection='3d')
surf3 = ax3.plot_surface(A, B, Loss_L2, cmap='viridis',
                         alpha=0.8, edgecolor='none')
ax3.scatter([a_opt_l2], [b_opt_l2], [Loss_L2[idx_min_l2 // 200, idx_min_l2 % 200]],
            color='red', s=100)
ax3.set_title(
    f'L2 Regularization (λ=2)\nOptimum: a={a_opt_l2:.2f}, b={b_opt_l2:.2f}', fontweight='bold')
ax3.set_xlabel('Parameter a')
ax3.set_ylabel('Parameter b')
ax3.set_zlabel('Loss')

plt.tight_layout()
plt.savefig('yaxb_3d_comparison_fixed.png', dpi=150, bbox_inches='tight')
plt.show()

# ============ 6. Plot 2: Contour Maps ============
fig, axes = plt.subplots(1, 3, figsize=(18, 5))

# Unregularized
ax = axes[0]
contour = ax.contourf(A, B, Loss_base, levels=30, cmap='viridis')
ax.plot(a_opt_base, b_opt_base, 'r*', markersize=20, label='Optimum')
ax.set_title('Unregularized', fontweight='bold')
ax.set_xlabel('Parameter a')
ax.set_ylabel('Parameter b')
ax.legend()
ax.grid(True, alpha=0.3)

# L1 Regularized
ax = axes[1]
contour = ax.contourf(A, B, Loss_L1, levels=30, cmap='viridis')
ax.plot(a_opt_l1, b_opt_l1, 'r*', markersize=20, label='Optimum')
# Draw L1 diamond contour
l1_val = 1.0
l1_x = np.linspace(-3, 3, 400)
l1_y_pos = l1_val - np.abs(l1_x)
l1_y_neg = -l1_val + np.abs(l1_x)
ax.plot(l1_x, l1_y_pos, 'yellow', linewidth=3,
        label='L1 contour', linestyle='--')
ax.plot(l1_x, l1_y_neg, 'yellow', linewidth=3, linestyle='--')
ax.axhline(0, color='white', linewidth=0.5, linestyle='--')
ax.axvline(0, color='white', linewidth=0.5, linestyle='--')
ax.set_title(
    f'L1 Regularization (λ=2)\na={a_opt_l1:.2f}, b={b_opt_l1:.2f}', fontweight='bold')
ax.set_xlabel('Parameter a')
ax.set_ylabel('Parameter b')
ax.legend()
ax.grid(True, alpha=0.3)

# L2 Regularized
ax = axes[2]
contour = ax.contourf(A, B, Loss_L2, levels=30, cmap='viridis')
ax.plot(a_opt_l2, b_opt_l2, 'r*', markersize=20, label='Optimum')
# Draw L2 circular contour
l2_r = np.sqrt(1.0)
circle = plt.Circle((0, 0), l2_r, color='yellow',
                    fill=False, linewidth=3, label='L2 contour')
ax.add_patch(circle)
ax.set_title(
    f'L2 Regularization (λ=2)\na={a_opt_l2:.2f}, b={b_opt_l2:.2f}', fontweight='bold')
ax.set_xlabel('Parameter a')
ax.set_ylabel('Parameter b')
ax.legend()
ax.grid(True, alpha=0.3)

plt.colorbar(contour, ax=axes, label='Loss')
plt.tight_layout()
plt.savefig('yaxb_contour_fixed.png', dpi=150, bbox_inches='tight')
plt.show()

# ============ 7. Plot 3: Parameter Shrinkage vs Lambda ============
lambda_vals = np.linspace(0, 4, 50)
a_l1_list, b_l1_list = [], []
a_l2_list, b_l2_list = [], []

for lam in lambda_vals:
    # L1
    Loss_L1_temp = compute_L1_loss(A, B, x, y, lambda_l1=lam)
    idx = np.argmin(Loss_L1_temp)
    a_l1_list.append(a_range[idx % 200])
    b_l1_list.append(b_range[idx // 200])

    # L2
    Loss_L2_temp = compute_L2_loss(A, B, x, y, lambda_l2=lam)
    idx = np.argmin(Loss_L2_temp)
    a_l2_list.append(a_range[idx % 200])
    b_l2_list.append(b_range[idx // 200])

fig, axes = plt.subplots(1, 2, figsize=(14, 5))

# L1 parameter curve
axes[0].plot(lambda_vals, a_l1_list, 'b-', linewidth=2, label='Parameter a')
axes[0].plot(lambda_vals, b_l1_list, 'r-', linewidth=2, label='Parameter b')
axes[0].set_title(
    'L1 Regularization: Parameters vs λ\n(Jumps to exactly 0)', fontweight='bold')
axes[0].set_xlabel('Regularization Strength λ')
axes[0].set_ylabel('Parameter Value')
axes[0].axhline(0, color='gray', linewidth=0.5, linestyle='--')
axes[0].legend()
axes[0].grid(True, alpha=0.3)

# L2 parameter curve
axes[1].plot(lambda_vals, a_l2_list, 'b-', linewidth=2, label='Parameter a')
axes[1].plot(lambda_vals, b_l2_list, 'r-', linewidth=2, label='Parameter b')
axes[1].set_title(
    'L2 Regularization: Parameters vs λ\n(Approaches 0 but never reaches)', fontweight='bold')
axes[1].set_xlabel('Regularization Strength λ')
axes[1].set_ylabel('Parameter Value')
axes[1].axhline(0, color='gray', linewidth=0.5, linestyle='--')
axes[1].legend()
axes[1].grid(True, alpha=0.3)

plt.tight_layout()
plt.savefig('yaxb_parameter_shrinkage_fixed.png', dpi=150, bbox_inches='tight')
plt.show()

# ============ 8. Plot 4: Gradient Descent Trajectory ============

def gradient_descent(a_init, b_init, x, y, lambda_val, reg_type='L1', lr=0.01, n_steps=100):
    """Gradient descent optimization trajectory"""
    a_traj = [a_init]
    b_traj = [b_init]

    a = a_init
    b = b_init

    for i in range(n_steps):
        # Compute gradient
        y_pred = a * x + b
        error = y_pred - y

        # Base gradient
        grad_a = 2 * np.mean(error * x)
        grad_b = 2 * np.mean(error)

        # Regularization gradient
        if reg_type == 'L1':
            grad_a += lambda_val * np.sign(a)
            grad_b += lambda_val * np.sign(b)
        else:  # L2
            grad_a += 2 * lambda_val * a
            grad_b += 2 * lambda_val * b

        # Update
        a -= lr * grad_a
        b -= lr * grad_b

        a_traj.append(a)
        b_traj.append(b)

    return np.array(a_traj), np.array(b_traj)

# Run gradient descent
a_init, b_init = 5.0, 3.0
a_L1_traj, b_L1_traj = gradient_descent(
    a_init, b_init, x, y, lambda_val=1.0, reg_type='L1', lr=0.005, n_steps=200)
a_L2_traj, b_L2_traj = gradient_descent(
    a_init, b_init, x, y, lambda_val=1.0, reg_type='L2', lr=0.005, n_steps=200)

fig, axes = plt.subplots(1, 2, figsize=(14, 5))

# L1 trajectory
axes[0].contourf(A, B, Loss_L1, levels=30, cmap='viridis', alpha=0.5)
axes[0].contour(A, B, Loss_L1, levels=10, colors='white',
                linewidths=0.5, alpha=0.5)
axes[0].plot(a_L1_traj, b_L1_traj, 'r-o', linewidth=2,
             markersize=3, label='Optimization Path')
axes[0].plot(a_L1_traj[-1], b_L1_traj[-1], 'r*', markersize=20,
             label=f'Final (a={a_L1_traj[-1]:.1f}, b={b_L1_traj[-1]:.1f})')
axes[0].axhline(0, color='yellow', linewidth=0.5, linestyle='--')
axes[0].axvline(0, color='yellow', linewidth=0.5, linestyle='--')
axes[0].set_title('L1 Regularization: GD Trajectory', fontweight='bold')
axes[0].set_xlabel('Parameter a')
axes[0].set_ylabel('Parameter b')
axes[0].legend()
axes[0].grid(True, alpha=0.3)

# L2 trajectory
axes[1].contourf(A, B, Loss_L2, levels=30, cmap='viridis', alpha=0.5)
axes[1].contour(A, B, Loss_L2, levels=10, colors='white',
                linewidths=0.5, alpha=0.5)
axes[1].plot(a_L2_traj, b_L2_traj, 'g-o', linewidth=2,
             markersize=3, label='Optimization Path')
axes[1].plot(a_L2_traj[-1], b_L2_traj[-1], 'g*', markersize=20,
             label=f'Final (a={a_L2_traj[-1]:.1f}, b={b_L2_traj[-1]:.1f})')
axes[1].set_title('L2 Regularization: GD Trajectory', fontweight='bold')
axes[1].set_xlabel('Parameter a')
axes[1].set_ylabel('Parameter b')
axes[1].legend()
axes[1].grid(True, alpha=0.3)

plt.tight_layout()
plt.savefig('yaxb_gd_trajectory_fixed.png', dpi=150, bbox_inches='tight')
plt.show()

# ============ 9. Print Summary ============
print("\n" + "=" * 60)
print("KEY CONCLUSIONS")
print("=" * 60)
print(f"""
1. Unregularized Optimum: a = {a_opt_base:.2f}, b = {b_opt_base:.2f}
   (Close to true values a={true_a}, b={true_b})

   → Parameter shrinks toward zero
   → Can become EXACTLY zero (SPARSITY!)
   → Geometric reason: Diamond has corners
   → Physical meaning: Feature selection

3. L2 Regularization (λ=2): a = {a_opt_l2:.2f}, b = {b_opt_l2:.2f}
   → Parameter shrinks but stays non-zero
   → Geometric reason: Circle is smooth
   → Physical meaning: Weight decay without elimination
""")
print("=" * 60)
```
